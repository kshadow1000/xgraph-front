cmake_minimum_required(VERSION 3.10)
project(xgraph-front)
if(NOT DEFINED CMAKE_BUILD_TYPE)
	set(CMAKE_BUILD_TYPE Release)
endif()
add_subdirectory(xgraph/expr)
set(CMAKE_C_FLAGS_RELEASE "-O3 -DNDEBUG=1")
set(CMAKE_C_FLAGS_DEBUG "-Og -g -DNDEBUG=0")
add_compile_options(-Wall -Werror)
add_link_options(-lm)
set(COMMON_SRCS "")
if(DEFINED LEAK)
	set(COMMON_SRCS ${COMMON_SRCS} xgraph/expr/leak.c)
endif()
if(DEFINED CMAKE_HOST_UNIX)
	add_subdirectory(xgraph/graph)
	execute_process(COMMAND sh -c "sh ${PROJECT_SOURCE_DIR}/systable.sh >${PROJECT_BINARY_DIR}/systable.c")
	include_directories(${PROJECT_BINARY_DIR})
	set(COMMON_SRCS ${COMMON_SRCS} common_symbols.c)
	add_executable(draw main.c ${COMMON_SRCS})
	add_executable(wave wave.c)
	add_executable(list list.c ${COMMON_SRCS})
	target_link_libraries(draw expr graph)
	target_link_libraries(wave expr)
	target_link_libraries(list expr)
	add_definitions(-DCOMMON_SYMBOLS)
endif()
add_executable(calc calc.c ${COMMON_SRCS})
target_link_libraries(calc expr)
